x = [1 3 4 2 5];
y = [1 2 3 4 5];
yr = y(end:-1:1);
r = length(x)+length(y)-1;
R = zeros(1,r);
for i=1:length(x)
    for j=1:length(y)
        if (i+j-1) <= r
            R(i+j-1) = R(i+j-1) + yr(j)*x(i);
        end
    end
end

R_func = xcorr(x,y);
R_conv = conv(x,yr);
stem(R);
figure;
stem(R_func);
figure;
stem(R_conv);
